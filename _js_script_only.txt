function doLogin() {
  const account = document.getElementById('login-name').value.trim().toLowerCase();
  const password = document.getElementById('login-pass').value;
  if(!account) { alert('请输入账号'); return; }
  if(!password) { alert('请输入密码'); return; }
  const acc = ACCOUNTS[account];
  if(!acc || acc.password !== password) { alert('账号或密码错误'); return; }
  currentUser = { name: acc.name, team: acc.team, isHost: acc.team === '__host__' };
  localStorage.setItem('fc26_user', JSON.stringify(currentUser));
  showLoginUI();
  document.querySelector('.nav button').click();
}

function doLogout() {
  currentUser = null;
  localStorage.removeItem('fc26_user');
  document.getElementById('login-overlay').classList.remove('hidden');
  document.getElementById('user-badge').style.display = 'none';
}

function showLoginUI() {
  if(!currentUser) {
    document.getElementById('login-overlay').classList.remove('hidden');
    document.getElementById('user-badge').style.display = 'none';
    return;
  }
  document.getElementById('login-overlay').classList.add('hidden');
  const badge = document.getElementById('user-badge');
  badge.style.display = 'inline-flex';
  document.getElementById('user-avatar').textContent = currentUser.name[0];
  document.getElementById('user-name').textContent = currentUser.name + (currentUser.isHost ? ' 👑' : ' · ' + currentUser.team);

  // Show/hide tabs based on role
  document.getElementById('nav-budget').style.display = '';
  document.getElementById('nav-awards').style.display = currentUser.isHost ? '' : 'none';
  document.getElementById('nav-settle').style.display = currentUser.isHost ? '' : 'none';
  document.getElementById('nav-admin').style.display = currentUser.isHost ? '' : 'none';

  // Re-init awards form when host logs in (dropdowns need currentUser context)
  if(currentUser.isHost) {
    initAwards();
    renderBudget();
  }

  // Render budget for the user's team
  renderBudget();
  updatePendingBadge();
}

// Team name mapping: TEAMS_DATA uses full names, PLAYER_TEAMS uses short names
const TEAM_NAME_MAP = { '曼彻斯特城': '曼城', '曼彻斯特联': '曼联' };
const TEAM_NAME_UNMAP = { '曼城': '曼彻斯特城', '曼联': '曼彻斯特联' };
function resolveTeamName(name) { return TEAM_NAME_MAP[name] || name; }
function fullTeamName(short) { return TEAM_NAME_UNMAP[short] || short; }

function getAllPlayers() {
  let players = [];
  TEAMS_DATA.forEach(t => {
    t.players.forEach(p => {
      players.push({ ...p, team: resolveTeamName(t.team), team_en: t.team_en, isPlayerTeam: t.is_player_team });
    });
  });
  return players;
}

function filterPlayers() {
  let players = getAllPlayers();
  const search = document.getElementById('search-input').value.toLowerCase();
  const pos = document.getElementById('filter-pos').value;
  const team = document.getElementById('filter-team').value;
  const ovrMin = parseInt(document.getElementById('filter-ovr-min').value) || 0;
  const ovrMax = parseInt(document.getElementById('filter-ovr-max').value) || 99;
  const priceMin = parseInt(document.getElementById('filter-price-min').value) || 0;

  if(search) players = players.filter(p => p.name.toLowerCase().includes(search) || p.name_en.toLowerCase().includes(search));
  if(pos) players = players.filter(p => p.position.includes(pos));
  if(team) players = players.filter(p => p.team === team);
  if(state.selectedTeam) players = players.filter(p => p.team === state.selectedTeam);
  players = players.filter(p => p.overall >= ovrMin && p.overall <= ovrMax);
  if(priceMin) players = players.filter(p => p.market_price >= priceMin);

  players.sort((a,b) => b.overall - a.overall);

  const tbody = document.getElementById('player-tbody');
  const myTeam = currentUser && !currentUser.isHost ? currentUser.team : null;

  tbody.innerHTML = players.map((p, pi) => {
    let tags = '';
    if(p.is_player) tags += '<span class="tag tag-player">玩家</span>';
    if(p.is_legend) tags += '<span class="tag tag-legend">传奇</span>';
    let rowClass = p.is_player ? 'player-row' : (p.is_legend ? 'legend-row' : '');
    const _idx = regPlayer(p, 'db_' + p.team + '_' + pi);
    
    // Action buttons in player database
    let actions = '<span style="color:var(--text2);font-size:11px;">—</span>';
    if(currentUser) {
      if(currentUser.isHost) {
        // Host has global control - show buy/sell for any team's players
        if(PLAYER_TEAMS.includes(p.team)) {
          actions = `<div style="display:flex;gap:3px;flex-wrap:wrap;">
            ${!p.is_player ? `<button class="btn btn-sm" style="font-size:10px;padding:2px 6px;color:var(--green);border-color:var(--green);" onclick="hostSellPlayer('${_idx}')">售AI +${p.sell_price}</button>` : ''}
            <button class="btn btn-sm" style="font-size:10px;padding:2px 6px;color:var(--accent);border-color:var(--accent);" onclick="hostBuyPlayer('${_idx}')">购入→选队</button>
            <button class="btn btn-sm" style="font-size:10px;padding:2px 6px;color:#a78bfa;border-color:#a78bfa;" onclick="quickP2PSell('${_idx}','${p.team}')">售玩家队</button>
          </div>`;
        } else {
          // AI team player - host can buy for any team
          actions = `<button class="btn btn-sm" style="font-size:10px;padding:2px 6px;color:var(--red);border-color:var(--red);" onclick="hostBuyPlayer('${_idx}')">购入→选队 -${p.market_price}</button>`;
        }
      } else if(myTeam && p.team !== myTeam && PLAYER_TEAMS.includes(myTeam)) {
        // Buy from other team
        if(PLAYER_TEAMS.includes(p.team)) {
          // From another player team - player-to-player trade
          actions = `<button class="btn btn-sm" style="color:var(--accent);border-color:var(--accent);font-size:11px;padding:2px 8px;" onclick="quickP2P('${_idx}','${myTeam}')">🔄 玩家交易</button>`;
        } else {
          // From AI team
          actions = `<button class="btn btn-sm" style="color:var(--red);border-color:var(--red);font-size:11px;padding:2px 8px;" onclick="quickBuy('${_idx}','${myTeam}')">购入 -${p.market_price}点</button>`;
        }
      } else if(myTeam && p.team === myTeam) {
        // Sell own team player
        actions = `<div style="display:flex;gap:4px;flex-wrap:wrap;">
          ${!p.is_player ? `<button class="btn btn-sm" style="color:var(--green);border-color:var(--green);font-size:10px;padding:2px 6px;" onclick="quickSell('${_idx}','${myTeam}')">售AI +${p.sell_price}</button>` : ''}
          <button class="btn btn-sm" style="color:var(--accent);border-color:var(--accent);font-size:10px;padding:2px 6px;" onclick="quickP2PSell('${_idx}','${myTeam}')">售给玩家队</button>
        </div>`;
      }
    }
    
    return `<tr class="${rowClass}"><td><strong>${p.name}</strong><br><span style="color:var(--text2);font-size:11px">${p.name_en}</span></td><td style="color:var(--accent)">${p.position}</td><td><strong>${p.overall}</strong></td><td style="color:var(--gold)">${p.market_price}点</td><td>${p.sell_price}点</td><td>${actions}</td><td>${tags}</td></tr>`;
  }).join('');

  document.getElementById('player-count').textContent = `显示 ${players.length} / ${getAllPlayers().length} 名球员`;
}

function swapPlayerFromSquad(pIdx) {
  if(state.selectedSlotIdx === null) {
    // No slot selected - select this player first (will be placed on next slot click)
    state._pendingPlayerIdx = pIdx;
    document.querySelectorAll('#squad-list .squad-item').forEach(el => el.style.background = '');
    document.querySelectorAll('#squad-list .squad-item')[pIdx].style.background = 'rgba(250,204,21,0.15)';
    updatePitchHint('已选中替补球员，点击场上位置放入');
    return;
  }
  
  // A slot is selected - place player there
  const slotIdx = state.selectedSlotIdx;
  const currentPIdx = state.pitchSlots[slotIdx].playerIdx;
  
  state.pitchSlots[slotIdx].playerIdx = pIdx;
  state.selectedSlotIdx = null;
  state._pendingPlayerIdx = null;
  applyFormation();
  updatePitchHint('');
}

function swapPlayerFromPitch(slotIdx) {
  if(state.selectedSlotIdx === null) {
    // Select this slot
    state.selectedSlotIdx = slotIdx;
    applyFormation();
    updatePitchHint('已选中场上球员，点击替补球员替换 / 点击另一个场上球员交换');
    return;
  }
  
  if(state.selectedSlotIdx === slotIdx) {
    // Deselect
    state.selectedSlotIdx = null;
    applyFormation();
    updatePitchHint('');
    return;
  }
  
  // Swap two pitch players
  const tmp = state.pitchSlots[state.selectedSlotIdx].playerIdx;
  state.pitchSlots[state.selectedSlotIdx].playerIdx = state.pitchSlots[slotIdx].playerIdx;
  state.pitchSlots[slotIdx].playerIdx = tmp;
  state.selectedSlotIdx = null;
  state._pendingPlayerIdx = null;
  applyFormation();
  updatePitchHint('');
}

function removePlayerFromPitch(slotIdx) {
  state.pitchSlots[slotIdx].playerIdx = null;
  state.selectedSlotIdx = null;
  state._pendingPlayerIdx = null;
  applyFormation();
  updatePitchHint('');
}

function updatePitchHint(text) {
  document.getElementById('pitch-hint').textContent = text;
}

function applyFormation() {
  const form = FORMATIONS[state.tacticsFormation];
  if(!form) return;
  document.getElementById('tactics-formation-label').textContent = state.tacticsFormation;
  
  // Save existing assignments
  const saved = state.pitchSlots.map(s => s ? s.playerIdx : null);
  
  state.pitchSlots = [];
  const container = document.getElementById('pitch-players');
  container.innerHTML = '';
  
  const color = TEAM_COLORS[state.tacticsTeam] || '#22d3ee';
  const squad = state.tacticsSquad || [];
  
  form.forEach((pos, i) => {
    let pIdx = null;
    if(saved[i] !== null && saved[i] !== undefined && squad[saved[i]]) {
      pIdx = saved[i];
    } else {
      const usedIdxs = new Set(state.pitchSlots.filter(s => s.playerIdx !== null).map(s => s.playerIdx));
      const autoP = squad.findIndex((p, idx) => !usedIdxs.has(idx));
      if(autoP >= 0) pIdx = autoP;
    }
    
    state.pitchSlots[i] = { x: pos[0], y: pos[1], playerIdx: pIdx };
    
    if(pIdx !== null && squad[pIdx]) {
      const player = squad[pIdx];
      const el = document.createElement('div');
      el.className = 'pitch-player' + (state.selectedSlotIdx === i ? ' selected' : '');
      el.style.left = pos[0] + '%';
      el.style.top = pos[1] + '%';
      el.style.background = color;
      el.onclick = () => swapPlayerFromPitch(i);
      el.innerHTML = `
        <div class="povr">${player.overall}</div>
        <div class="ppos">${player.position}</div>
        <div class="pname">${player.name}</div>
        <div class="pprice">${player.market_price}点</div>
      `;
      
      // Right-click to remove
      el.addEventListener('contextmenu', (e) => { e.preventDefault(); removePlayerFromPitch(i); });
      
      container.appendChild(el);
    } else {
      const slot = document.createElement('div');
      slot.className = 'pitch-slot' + (state.selectedSlotIdx === i ? ' highlight' : '');
      slot.style.left = pos[0] + '%';
      slot.style.top = pos[1] + '%';
      slot.textContent = '+';
      slot.onclick = () => {
        if(state._pendingPlayerIdx !== null && state._pendingPlayerIdx !== undefined) {
          state.pitchSlots[i].playerIdx = state._pendingPlayerIdx;
          state._pendingPlayerIdx = null;
          state.selectedSlotIdx = null;
          applyFormation();
          updatePitchHint('');
        } else {
          swapPlayerFromPitch(i);
        }
      };
      container.appendChild(slot);
    }
  });
  
  // Update squad list dimming
  const onPitchIdxs = new Set(state.pitchSlots.filter(s => s.playerIdx !== null).map(s => s.playerIdx));
  document.querySelectorAll('#squad-list .squad-item').forEach(el => {
    const idx = parseInt(el.dataset.playerIdx);
    el.classList.toggle('on-pitch', onPitchIdxs.has(idx));
    el.style.background = '';
  });
}

function clearPitch() {
  if(state.pitchSlots) state.pitchSlots.forEach(s => s.playerIdx = null);
  state.selectedSlotIdx = null;
  state._pendingPlayerIdx = null;
  applyFormation();
  updatePitchHint('');
}

// ===== MATCH SIMULATOR =====
// Formation matchup table (real football tactical relationships)
const FORMATION_MATCHUP = {
  '4-3-3':  { strong: ['4-4-2','5-3-2','3-4-3'], weak: ['4-2-3-1','4-5-1','3-5-2'] },
  '4-4-2':  { strong: ['4-3-3','3-5-2','4-1-4-1'], weak: ['4-5-1','4-2-3-1','3-4-3'] },
  '4-2-3-1':{ strong: ['4-3-3','3-5-2','4-4-2'], weak: ['4-4-1-1','5-4-1','4-1-4-1'] },
  '4-5-1':  { strong: ['4-4-2','4-3-3','3-4-3'], weak: ['4-2-3-1','4-3-2-1','5-3-2'] },
  '3-5-2':  { strong: ['4-3-3','4-4-2','4-5-1'], weak: ['4-2-3-1','5-3-2','4-4-1-1'] },
  '3-4-3':  { strong: ['4-5-1','5-3-2','4-4-2'], weak: ['4-3-3','4-2-3-1','3-5-2'] },
  '5-3-2':  { strong: ['4-3-3','3-5-2','4-4-2'], weak: ['4-2-3-1','3-4-3','4-1-4-1'] },
  '5-4-1':  { strong: ['4-3-3','4-2-3-1','3-4-3'], weak: ['4-4-2','3-5-2','4-5-1'] },
  '4-1-4-1':{ strong: ['4-2-3-1','3-5-2','5-3-2'], weak: ['4-4-2','4-5-1','3-4-3'] },
  '4-4-1-1':{ strong: ['4-2-3-1','5-3-2','3-5-2'], weak: ['4-3-3','4-4-2','4-5-1'] },
  '4-3-2-1':{ strong: ['4-4-2','4-5-1','3-4-3'], weak: ['4-3-3','4-2-3-1','5-4-1'] }
};

function getFormationBonus(myForm, oppForm) {
  const m = FORMATION_MATCHUP[myForm];
  if(!m) return 0;
  if(m.strong && m.strong.includes(oppForm)) return 3;
  if(m.weak && m.weak.includes(oppForm)) return -3;
  return 0;
}

function getPositionBonus(squad, formation) {
  let bonus = 0;
  squad.forEach(p => {
    const pos = p.position;
    // Position-formation synergy
    if(formation.includes('5') && (pos==='LM'||pos==='RM'||pos==='LW'||pos==='RW'||pos==='LWB'||pos==='RWB')) bonus += 0.5;
    if(formation.includes('3') && (pos==='CB'||pos==='CDM')) bonus += 0.3;
    if((formation==='4-3-3'||formation==='4-2-3-1') && (pos==='CAM'||pos==='CF'||pos==='LW'||pos==='RW')) bonus += 0.5;
    if(formation==='4-4-2' && (pos==='ST'||pos==='CM')) bonus += 0.3;
    // Attacker bonus for goalscoring potential
    if(pos==='ST'||pos==='CF'||pos==='CAM') bonus += 0.3;
    if(pos==='GK') bonus += 0.2; // GK contributes to defense
  });
  return bonus;
}

function simulateMatch() {
  const homeName = document.getElementById('match-home').value;
  const awayName = document.getElementById('match-away').value;
  if(homeName === awayName) { alert('请选择不同的球队！'); return; }
  
  const homeForm = document.getElementById('match-home-formation').value;
  const awayForm = document.getElementById('match-away-formation').value;
  
  const homeSquad = getMatchSquad('home');
  const awaySquad = getMatchSquad('away');
  
  // Calculate base strengths
  const homeAvg = homeSquad.reduce((s,p) => s+p.overall, 0) / 11;
  const awayAvg = awaySquad.reduce((s,p) => s+p.overall, 0) / 11;
  
  // Formation bonuses
  const homeFormBonus = getFormationBonus(homeForm, awayForm);
  const awayFormBonus = getFormationBonus(awayForm, homeForm);
  
  // Position-formation synergy
  const homePosBonus = getPositionBonus(homeSquad, homeForm);
  const awayPosBonus = getPositionBonus(awaySquad, awayForm);
  
  // Calculate effective strength
  const homeEff = homeAvg + homeFormBonus + homePosBonus + (Math.random()*12 - 6);
  const awayEff = awayAvg + awayFormBonus + awayPosBonus + (Math.random()*12 - 6);
  
  // Determine expected goals (based on effective strength ratio)
  const homeProb = homeEff / (homeEff + awayEff);
  const awayProb = awayEff / (homeEff + awayEff);
  
  // Generate goals using Poisson-like distribution
  const homeExpected = Math.max(0.5, homeProb * 3.2 + (Math.random()*1.5 - 0.75));
  const awayExpected = Math.max(0.5, awayProb * 3.2 + (Math.random()*1.5 - 0.75));
  
  let homeGoals = 0, awayGoals = 0;
  for(let i = 0; i < 10; i++) { if(Math.random() < homeExpected/10) homeGoals++; }
  for(let i = 0; i < 10; i++) { if(Math.random() < awayExpected/10) awayGoals++; }
  
  // Generate match events
  const events = [];
  
  // Goal events
  for(let g = 0; g < homeGoals; g++) {
    const minute = Math.floor(Math.random()*90) + 1;
    // Attackers more likely to score
    const attackers = homeSquad.filter(p => ['ST','CF','CAM','LW','RW'].includes(p.position));
    const scorers = attackers.length > 0 ? attackers : homeSquad;
    const scorer = scorers[Math.floor(Math.random()*scorers.length)];
    events.push({ min:minute, type:'goal', side:'home', team:homeName, player:scorer.name, ovr:scorer.overall });
  }
  for(let g = 0; g < awayGoals; g++) {
    const minute = Math.floor(Math.random()*90) + 1;
    const attackers = awaySquad.filter(p => ['ST','CF','CAM','LW','RW'].includes(p.position));
    const scorers = attackers.length > 0 ? attackers : awaySquad;
    const scorer = scorers[Math.floor(Math.random()*scorers.length)];
    events.push({ min:minute, type:'goal', side:'away', team:awayName, player:scorer.name, ovr:scorer.overall });
  }
  
  // Yellow cards (random)
  const cardCount = Math.floor(Math.random()*4);
  for(let i = 0; i < cardCount; i++) {
    const minute = Math.floor(Math.random()*90) + 1;
    const isHome = Math.random() < 0.5;
    const team = isHome ? homeSquad : awaySquad;
    const player = team[Math.floor(Math.random()*team.length)];
    events.push({ min:minute, type:'card', side:isHome?'home':'away', team:isHome?homeName:awayName, player:player.name, ovr:player.overall });
  }
  
  // Halftime / Fulltime markers
  events.push({ min:45, type:'whistle', side:'neutral', text:'半场' });
  events.push({ min:90, type:'whistle', side:'neutral', text:'全场结束' });
  
  events.sort((a,b) => a.min - b.min);
  
  // Build analysis text
  let analysis = '';
  if(homeFormBonus > 0) analysis += `<strong>${homeName}</strong>（${homeForm}）阵型克制 <strong>${awayName}</strong>（${awayForm}）+${homeFormBonus} 战力<br>`;
  else if(homeFormBonus < 0) analysis += `<strong>${awayName}</strong>（${awayForm}）阵型克制 <strong>${homeName}</strong>（${homeForm}）+${Math.abs(homeFormBonus)} 战力<br>`;
  else analysis += `双方阵型无克制关系<br>`;
  analysis += `综合战力: <strong style="color:${TEAM_COLORS[homeName]||'var(--accent)'}">${homeName}</strong> ${homeEff.toFixed(1)} vs <strong style="color:${TEAM_COLORS[awayName]||'var(--accent2)'}">${awayName}</strong> ${awayEff.toFixed(1)}`;
  
  // Build mini pitch for result display
  const homeFormPos = FORMATIONS[homeForm] || FORMATIONS['4-3-3'];
  const awayFormPos = FORMATIONS[awayForm] || FORMATIONS['4-3-3'];
  const homeColor = TEAM_COLORS[homeName] || '#22d3ee';
  const awayColor = TEAM_COLORS[awayName] || '#f472b6';
  
  function miniPitchHTML(squad, positions, color, isHome) {
    return positions.map((pos, i) => {
      const p = squad[i];
      if(!p) return '';
      const y = isHome ? pos[1]*0.45 : 55 + pos[1]*0.45;
      return `<div class="mini-player" style="left:${pos[0]}%;top:${y}%"><span class="mini-ovr">${p.overall}</span><span class="mini-name">${p.name}</span></div>`;
    }).join('');
  }
  
  // Result text
  let resultText = '平局';
  let resultEmoji = '🤝';
  if(homeGoals > awayGoals) { resultText = homeName + ' 获胜！'; resultEmoji = '🏆'; }
  else if(awayGoals > homeGoals) { resultText = awayName + ' 获胜！'; resultEmoji = '🏆'; }
  
  // Timeline
  function eventHTML(e) {
    if(e.type === 'whistle') return `<div class="match-event neutral">⏱️ ${e.min}' — ${e.text}</div>`;
    if(e.type === 'goal') {
      const icon = e.side === 'home' ? '⚽' : '⚽';
      return `<div class="match-event ${e.side}">${icon} ${e.min}' <strong>${e.player}</strong> <span style="opacity:0.6">(OVR ${e.ovr})</span></div>`;
    }
    if(e.type === 'card') {
      return `<div class="match-event ${e.side}">🟨 ${e.min}' ${e.player}</div>`;
    }
    return '';
  }
  
  document.getElementById('match-result-area').innerHTML = `
    <div class="card match-result">
      <div style="font-size:13px;color:var(--text2);margin-bottom:6px;">${homeName} (${homeForm}) vs ${awayName} (${awayForm})</div>
      <div class="match-score"><span style="color:${homeColor}">${homeGoals}</span> - <span style="color:${awayColor}">${awayGoals}</span></div>
      <div style="color:var(--gold);font-size:16px;margin:6px 0;">${resultEmoji} ${resultText}</div>
      <div class="match-analysis">${analysis}</div>
      <div style="display:flex;gap:10px;justify-content:center;margin:10px 0;">
        <div class="pitch-mini">${miniPitchHTML(homeSquad, homeFormPos, homeColor, true)}</div>
        <div class="pitch-mini">${miniPitchHTML(awaySquad, awayFormPos, awayColor, false)}</div>
      </div>
      <div style="font-size:13px;font-weight:600;margin:12px 0 6px;">📋 比赛事件</div>
      <div class="match-timeline">${events.map(eventHTML).join('')}</div>
    </div>
  `;
}

function submitAward(pts, label) {
  if(!currentUser || !currentUser.isHost) { alert('仅主播可录入赛事奖项！'); return; }
  const team = document.getElementById('award-team').value;
  const season = document.getElementById('award-season').value;
  if(!team) { alert('请选择获奖球队'); return; }
  
  const award = { season: season, team: team, label: label, pts: pts, time: new Date().toLocaleString('zh-CN') };
  state.awards.unshift(award);
  
  // Add points to team account
  state.budgets[team].balance += pts;
  state.budgets[team].log.unshift({ action: `🏆 ${season}-${Number(season)+1} ${label}`, pts: '+'+pts, time: new Date().toLocaleString('zh-CN') });
  
  saveState();
  renderAwards();
  renderBudget();
  updatePendingBadge();
}

function renderAwards() {
  const historyEl = document.getElementById('awards-history');

        </div>

function deleteAward(idx) {
  const a = state.awards[idx];
  if(!a) return;
  if(!confirm(`确定删除：${a.team} ${a.season}-${Number(a.season)+1} ${a.label}？\n将扣除对应 ${a.pts} 点数。`)) return;
  
  // Deduct points from team
  state.budgets[a.team].balance -= a.pts;
  state.budgets[a.team].log.unshift({ action: `🗑️ 撤回 ${a.season}-${Number(a.season)+1} ${a.label}`, pts: '-'+a.pts, time: new Date().toLocaleString('zh-CN') });
  
  state.awards.splice(idx, 1);
  saveState();
  renderAwards();
  renderBudget();
  updatePendingBadge();
}

function clearAwards() {
  if(!confirm('确定清空所有奖项记录？对应点数变动将通过撤销记录体现。')) return;
  state.awards = [];
  saveState();
  renderAwards();
}

function renderAwardSeasonSummary() {
  const summaryEl = document.getElementById('award-season-summary');
  if(!state.awards || state.awards.length === 0) { summaryEl.innerHTML = ''; return; }
  
  // Group by season
  const bySeason = {};
  state.awards.forEach(a => {
    if(!bySeason[a.season]) bySeason[a.season] = {};
    if(!bySeason[a.season][a.team]) bySeason[a.season][a.team] = 0;
    bySeason[a.season][a.team] += a.pts;
  });
  
  const seasons = Object.keys(bySeason).sort();
  summaryEl.innerHTML = `<div class="card"><div class="card-title">📊 各赛季奖项点数汇总</div>` +
    seasons.map(s => {
      const teams = bySeason[s];
      return `<div style="margin-bottom:12px;"><div style="font-weight:600;font-size:14px;color:var(--accent);margin-bottom:6px;">${s}-${Number(s)+1} 赛季</div>` +
        Object.entries(teams).map(([team, pts]) => {
          const color = TEAM_COLORS[team] || 'var(--text)';
          return `<div style="display:flex;justify-content:space-between;padding:4px 0;font-size:13px;border-bottom:1px solid var(--border);">
            <span style="color:${color}">${team}</span><span style="color:var(--gold);font-weight:600;">+${pts}点</span>
          </div>`;
        }).join('') + '</div>';
    }).join('') + '</div>';
}

  });

    btn.addEventListener('click', function() { doPlayerUpgrade(this.dataset.upgradeOvr, this.dataset.team, 'ovr'); });
  });

  });

function filterTransferList(search, type, team) {
  var b = state.budgets[team];
  var players = getAllPlayers().filter(function(p) { return p.team !== team; }).sort(function(a,b) { return b.overall - a.overall; });
  if(search) players = players.filter(function(p) { return p.name.toLowerCase().includes(search.toLowerCase()); });
  players = players.filter(function(p) { return p.market_price <= b.balance; });
  
  var listEl = document.getElementById('transfer-list');
  var buyhtml = '';
  players.slice(0,50).forEach(function(p, pi) {
    var bidx = regPlayer(p, 'buy_' + team + '_' + pi);
    buyhtml += '<div style="display:flex;align-items:center;gap:8px;padding:8px;border-bottom:1px solid var(--border);font-size:13px;">';
    buyhtml += '<span style="color:var(--accent);font-weight:600;width:35px">' + p.position + '</span>';
    buyhtml += '<span style="flex:1">' + p.name + ' (' + p.team + ') <span style="color:var(--gold)">' + p.overall + '</span></span>';
    buyhtml += '<button class="btn btn-sm" style="color:var(--red)" data-exec-buy="' + bidx + '" data-exec-team="' + team + '">购入 -' + p.market_price + '点</button>';
    buyhtml += '</div>';
  });
  if(!buyhtml) buyhtml = '<div style="color:var(--text2);padding:10px;font-size:13px;">无匹配球员或余额不足</div>';
  listEl.innerHTML = buyhtml;
  listEl.querySelectorAll('[data-exec-buy]').forEach(function(btn) {
    btn.addEventListener('click', function() {
      executeBuy(this.getAttribute('data-exec-team'), this.getAttribute('data-exec-buy'));
    });
  });
}

function executeBuy(team, playerIdx) {
  const p = clickPlayer(playerIdx);
  if(!p) return;
  const b = state.budgets[team];
  if(b.balance < p.market_price) { alert('余额不足！'); return; }
  if(b.tradeUsed >= 5) { alert('本赛季交易额度已用完！'); return; }
  
  // Create pending operation instead of direct execution
  const op = {
    type: 'buy', team: team, player: { name:p.name, position:p.position, overall:p.overall, team:p.team },
    action: `购入 ${p.name}`, pts: -p.market_price,
    player_name: p.name, time: new Date().toLocaleString('zh-CN'),
    submitted_by: currentUser ? currentUser.name : 'unknown'
  };
  state.pendingOps.push(op);
  saveState(); renderBudget(); closeModal(); updatePendingBadge();
}

function submitP2PSell(playerIdx, sellerTeam) {
  const p = clickPlayer(playerIdx);
  if(!p) return;
  const buyerTeam = document.getElementById('p2p-buyer-team').value;
  const buyerPts = parseInt(document.getElementById('p2p-buyer-pts').value) || 0;
  const sellerPts = parseInt(document.getElementById('p2p-seller-pts').value) || 0;
  
  if(buyerPts < 0 || sellerPts < 0) { alert('点数不能为负数！'); return; }
  
  const buyerBudget = state.budgets[buyerTeam];
  if(buyerBudget.tradeUsed >= 5) { alert(buyerTeam + ' 交易额度已用完！'); return; }
  if(buyerBudget.balance < buyerPts) { alert(buyerTeam + ' 余额不足！'); return; }

    type: 'p2p_buy', team: buyerTeam, player: { name:p.name, position:p.position, overall:p.overall, fromTeam: sellerTeam },

    type: 'p2p_sell', team: sellerTeam, player: { name:p.name, position:p.position, overall:p.overall, toTeam: buyerTeam },
    action: `玩家交易售出 ${p.name}（至${buyerTeam}）`, pts: '+'+sellerPts,

    submitted_by: currentUser ? currentUser.name : 'unknown'

  state.pendingOps.push(opSell);

}

function showBteamModal(team) {
  if(!canOperate(team)) { alert('你只能操作自己所属的球队！'); return; }
  const b = state.budgets[team];
  if(b.bteamUsed >= 5) { alert('本赛季B队额度已用完！'); return; }
  if(b.balance < 2) { alert('余额不足（需2点）！'); return; }
  const teamData = TEAMS_DATA.find(t => t.team === fullTeamName(team));
  const players = teamData.players.filter(p => !p.is_player).sort((a,b) => b.overall - a.overall);
  
  document.getElementById('transfer-title').textContent = `⬇️ ${team} - 下放球员到B队`;
  document.getElementById('transfer-body').innerHTML = `
    <div style="margin-bottom:10px;font-size:13px;color:var(--text2)">消耗2点 | 剩余额度: <strong style="color:var(--green)">${5-b.bteamUsed}</strong>/5</div>
    <div style="max-height:300px;overflow-y:auto;">
      ${players.map((p, pi) => `
        <div style="display:flex;align-items:center;gap:8px;padding:8px;border-bottom:1px solid var(--border);font-size:13px;">
          <span style="color:var(--accent);font-weight:600;width:35px">${p.position}</span>

        </div>
      `).join('')}
    </div>
  `;
  document.getElementById('transfer-modal').classList.add('show');
}

function showRecallModal(team) {
  if(!canOperate(team)) { alert('你只能操作自己所属的球队！'); return; }
  showBteamModal(team);
}

function demotePlayer(team, playerIdx) {
  const p = clickPlayer(playerIdx);
  if(!p) return;

  if(b.bteamUsed >= 5) { alert('额度已用完！'); return; }
  if(b.balance < 2) { alert('余额不足！'); return; }
  
  const op = {
    type: 'demote', team: team, player: { name:p.name, position:p.position, overall:p.overall },
    action: `下放 ${p.name} 到B队`, pts: '-2',

  };

  saveState(); renderBudget(); renderBteam(); closeModal(); updatePendingBadge();
}

  const op = {
    type: 'recall', team: team, player: { name:p.name, position:p.position, overall:p.overall },
    action: `召回 ${p.name} 从B队`, pts: '-2',
    player_name: p.name, time: new Date().toLocaleString('zh-CN'),
    submitted_by: currentUser ? currentUser.name : 'unknown'

}

        </div>` : ''}
      </span>

      ${isSellToAI ? `

      ` : `

    </div>`;
  }).join('');
}

function executePendingOp(op) {
  const b = state.budgets[op.team];
  if(!b) return;
  
  const v = parseInt(op.pts) || 0;
  b.balance += v;
  
  if(op.type === 'buy') {
    b.tradeUsed++;
  } else if(op.type === 'sell') {
    b.tradeUsed++;
  } else if(op.type === 'p2p_buy') {
    b.tradeUsed++;
  } else if(op.type === 'p2p_sell') {
    // Player-to-player sell: no tradeUsed for seller (negotiated directly)
  } else if(op.type === 'upgrade') {
    // Upgrade: just deduct points, no quota usage
  } else if(op.type === 'demote') {
    b.bteamUsed++;
    if(op.player) state.bteams[op.team].demoted.push(op.player);
  } else if(op.type === 'recall') {
    b.bteamUsed++;
    if(op.player) {
      const idx = state.bteams[op.team].demoted.findIndex(p => p.name === op.player.name);
      if(idx >= 0) state.bteams[op.team].demoted.splice(idx, 1);
    }
  }
  
  b.log.unshift({ action: op.action, pts: op.pts, time: new Date().toLocaleString('zh-CN') + ' ✓' });
}

function approveSellPending(idx) {
  const op = state.pendingOps[idx];
  if(!op || op.type !== 'sell') return;
  const destSelect = document.getElementById('sell-dest-' + idx);
  const destTeam = destSelect ? destSelect.value : '';
  if(!destTeam) { alert('请先选择球员去向球队！'); return; }
  if(!confirm(`确认：${op.player_name || op.action}\n去向：${destTeam === '__free_agent__' ? '自由球员' : destTeam}`)) return;
  // Move player in TEAMS_DATA
  if(destTeam !== '__free_agent__') {
    movePlayerInData(op.player_name, op.team, destTeam);
  } else {
    movePlayerInData(op.player_name, op.team, null);
  }
  executePendingOp(op);
  state.pendingOps.splice(idx, 1);
  saveState(); renderPendingOps(); renderBudget(); renderBteam(); renderSettlement(); renderQuickAccount(); updatePendingBadge();
}

function movePlayerInData(playerName, fromTeam, toTeam) {
  // Find player in TEAMS_DATA and move to destination team
  const srcTeam = TEAMS_DATA.find(t => t.team === fullTeamName(fromTeam));
  if(!srcTeam) return;
  const pIdx = srcTeam.players.findIndex(p => p.name === playerName);
  if(pIdx < 0) return;
  const player = srcTeam.players.splice(pIdx, 1)[0];
  if(toTeam) {
    const destTeam = TEAMS_DATA.find(t => t.team === fullTeamName(toTeam));
    if(destTeam) destTeam.players.push(player);
  }
}

// ===== PLAYER UPGRADE SYSTEM =====
function doPlayerUpgrade(playerIdx, team, type) {
  var p = clickPlayer(playerIdx);
  if(!p) return;
  var b = state.budgets[team];
  if(b.balance < 1) { alert('余额不足！升级需要至少1点。'); return; }
  var typeName = type === 'ovr' ? '能力' : '潜力';
  var costStr = '1';
  
  var qtyStr = prompt(
    '⚡ ' + p.name + '（' + p.position + ' OVR ' + p.overall + '）\n' +
    '当前' + typeName + (type === 'pot' ? '加成: ' + (p.potential_bump || '无') : '') + '\n\n' +
    '输入要消耗的点数（每1点 = ' + typeName + '+1）：',
    '1'
  );
  if(!qtyStr) return;
  var qty = parseInt(qtyStr);
  if(isNaN(qty) || qty < 1) { alert('请输入有效的正整数！'); return; }
  if(qty > b.balance) { alert('余额不足！当前仅 ' + b.balance + ' 点，需要 ' + qty + ' 点。'); return; }
  if(!confirm('确认：' + p.name + ' ' + typeName + ' +' + qty + '，消耗 ' + qty + ' 点？')) return;
  
  // Apply upgrade
  if(type === 'ovr') {
    p.overall += qty;
    // Update sell_price and market_price based on new overall
    var prices = getPrices(p.overall);
    p.sell_price = prices.sell;
    p.market_price = prices.market;
  } else {
    if(!p.potential_bump) p.potential_bump = '';
    for(var i = 0; i < qty; i++) p.potential_bump += '+1';
  }
  
  // Record in log
  var op = {
    type: 'upgrade', team: team,
    player: { name: p.name, position: p.position, overall: p.overall },
    action: p.name + ' ' + typeName + '+' + qty,
    pts: '-' + qty,
    player_name: p.name, time: new Date().toLocaleString('zh-CN'),
    submitted_by: currentUser ? currentUser.name : 'unknown'
  };
  state.pendingOps.push(op);
  saveState(); renderBudget(); renderQuickAccount(); updatePendingBadge();
}

function getPrices(overall) {
  var map = {99:[42,14],98:[40,13],97:[38,12],96:[36,12],95:[34,11],94:[32,10],93:[30,10],92:[28,9],91:[26,8],90:[24,8],89:[22,7]};
  if(map[overall]) return {market:map[overall][0], sell:map[overall][1]};
  if(overall >= 85) return {market:20, sell:6};
  if(overall >= 80) return {market:10, sell:3};
  if(overall >= 75) return {market:5, sell:2};
  return {market:3, sell:1};
}

function renderSettlement(filter) {
  filter = filter || 'all';
  const container = document.getElementById('settle-content');
  let html = '';

  PLAYER_TEAMS.forEach(team => {
    const b = state.budgets[team];
    const bt = state.bteams[team];
    if(!b || !b.log || b.log.length === 0) return;

    let items = b.log.filter(log => {
      if(filter === 'all') return true;
      if(filter === 'trade') return log.action.includes('购入') || log.action.includes('出售');
      if(filter === 'bteam') return log.action.includes('下放') || log.action.includes('召回');
      if(filter === 'award') return log.action.includes('冠军') || log.action.includes('季军') || log.action.includes('亚军');
      return true;
    });

    if(items.length === 0) return;

    html += `<div class="settle-card">
      <div class="settle-team"><span class="dot" style="background:${TEAM_COLORS[team]}"></span>${team}</div>`;

    items.forEach(item => {
      const isPlus = item.pts.startsWith('+');
      const cls = isPlus ? 'plus' : 'minus';
      let icon = '💰';
      if(item.action.includes('冠军')) icon = '🏆';
      else if(item.action.includes('季军') || item.action.includes('亚军')) icon = '🏅';
      else if(item.action.includes('下放')) icon = '⬇️';
      else if(item.action.includes('召回')) icon = '⬆️';
      else if(item.action.includes('购入')) icon = '📥';
      else if(item.action.includes('出售')) icon = '📤';

      html += `<div class="settle-item">
        <span class="si-icon">${icon}</span>
        <span class="si-detail">${item.action}<br><span style="font-size:11px;color:var(--text2)">${item.time}</span></span>
        <span class="si-pts ${cls}">${item.pts}</span>
      </div>`;
    });

    // B队下放球员清单
    if((filter === 'all' || filter === 'bteam') && bt.demoted && bt.demoted.length > 0) {
      html += `<div style="margin-top:8px;padding:8px 12px;background:var(--bg);border-radius:8px;font-size:12px;color:var(--text2);">
        <strong style="color:var(--text)">B队(${bt.name || '未设置'})在册球员：</strong><br>`;
      bt.demoted.forEach(p => {
        html += `<span style="color:var(--accent)">${p.position}</span> ${p.name} <span style="color:var(--gold)">${p.overall}</span>　`;
      });
      html += `</div>`;
    }

    html += `</div>`;
  });

  if(!html) {
    html = '<div class="settle-empty">📭 当前赛季暂无操作记录</div>';
  }

  container.innerHTML = html;
}

function calcMarketPrice(ovr) {
  if(ovr>=99) return 42; if(ovr>=98) return 40; if(ovr>=97) return 38; if(ovr>=96) return 36; if(ovr>=95) return 34;
  if(ovr>=94) return 32; if(ovr>=93) return 30; if(ovr>=92) return 28; if(ovr>=91) return 26; if(ovr>=90) return 24;
  if(ovr>=89) return 22; if(ovr>=88) return 20; if(ovr>=85) return 20; if(ovr>=84) return 10; if(ovr>=80) return 10;
  if(ovr>=79) return 5; if(ovr>=75) return 5; return 3;
}
function calcSellPrice(ovr) {
  if(ovr>=95) return Math.round(calcMarketPrice(ovr)*0.33);
  if(ovr>=88) return 6; if(ovr>=85) return 6; if(ovr>=80) return 3; if(ovr>=75) return 2; return 1;
}

function adminSearchPlayer() {
  const keyword = document.getElementById('admin-player-search').value.trim().toLowerCase();
  const teamFilter = document.getElementById('admin-player-team-filter').value;
  const container = document.getElementById('admin-player-result');
  if(!keyword) { container.innerHTML = '搜索球员进行修改'; return; }

  const allP = getAllPlayers();
  const results = allP.filter(p => {
    const matchName = p.name.toLowerCase().includes(keyword) || (p.name_en && p.name_en.toLowerCase().includes(keyword));
    const matchTeam = !teamFilter || p.team === teamFilter;
    return matchName && matchTeam;
  }).slice(0, 10);

  if(!results.length) { container.innerHTML = '未找到球员'; return; }
  container.innerHTML = results.map(p => {
    const ovrColor = p.overall >= 90 ? 'var(--gold)' : p.overall >= 80 ? 'var(--green)' : p.overall >= 70 ? 'var(--accent)' : 'var(--text2)';
    return `<div class="admin-player-card">

      <div class="player-info">
        <div class="pname">${p.name}${p.name_en ? ' ('+p.name_en+')' : ''}</div>
        <div class="pmeta">${p.team} · ${p.position} · 购入${p.market_price}点 / 售AI${p.sell_price}点</div>
      </div>
      <input class="admin-ovr-input" type="number" min="40" max="99" value="${p.overall}" id="admin-ovr-${p.team}-${p.name_en || p.name}">
      <button class="btn btn-sm" style="background:var(--gold);color:#000;" onclick="adminSetOverall('${p.team}','${(p.name_en||p.name).replace(/'/g,"\\'")}')">修改</button>
    </div>`;
  }).join('');
}

function adminSetOverall(team, playerKey) {
  const input = document.getElementById('admin-ovr-' + team + '-' + playerKey);
  if(!input) return;
  const newOvr = parseInt(input.value);
  if(isNaN(newOvr) || newOvr < 40 || newOvr > 99) { alert('能力值需在 40-99 之间'); return; }

  // Find and update in TEAMS_DATA
  let playerName = '';
  TEAMS_DATA.forEach(t => {
    if(t.team === fullTeamName(team)) {
      t.players.forEach(p => {
        const key = p.name_en || p.name;
        if(key === playerKey) {
          const oldOvr = p.overall;
          p.overall = newOvr;
          // Recalculate prices
          p.market_price = calcMarketPrice(newOvr);
          p.sell_price = calcSellPrice(newOvr);
          playerName = p.name;
          adminAddLog(`🎯 修改能力值：${playerName}（${team}）${oldOvr} → ${newOvr}，购入${p.market_price}/售AI${p.sell_price}`);
        }
      });
    }
  });

  if(playerName) {
    alert(`✅ ${playerName} 能力值已修改为 ${newOvr}`);
    adminSearchPlayer(); // refresh
  }
}

function adminAdjustBudget() {
  if(!currentUser || !currentUser.isHost) { alert('仅主播可操作！'); return; }
  const team = document.getElementById('admin-budget-team').value;
  const action = document.getElementById('admin-budget-action').value;
  const pts = parseInt(document.getElementById('admin-budget-pts').value);
  if(isNaN(pts) || pts < 0) { alert('请输入有效的点数'); return; }

  const s = JSON.parse(localStorage.getItem('fc26_state') || '{}');
  if(!s.budgets) s.budgets = {};
  if(!s.budgets[team]) s.budgets[team] = { pts: 0, log: [] };

  const oldPts = s.budgets[team].pts;
  let newPts;
  let actionText;
  if(action === 'set') { newPts = pts; actionText = `设为`; }
  else if(action === 'add') { newPts = oldPts + pts; actionText = `增加`; }
  else { newPts = Math.max(0, oldPts - pts); actionText = `减少`; }

  s.budgets[team].pts = newPts;
  s.budgets[team].log.push(`${actionText} ${pts}点 → ${newPts}点（管理员操作）`);
  localStorage.setItem('fc26_state', JSON.stringify(s));

  adminAddLog(`💰 调整点数：${team} ${oldPts}点 → ${newPts}点（${actionText}${pts}）`);
  updateAdminBudgetStatus();
  renderBudget();
  alert(`✅ ${team} 点数已${actionText} ${pts}，当前余额：${newPts}点`);
}

function adminMovePlayer(fromTeam, playerKey) {
  const selId = 'admin-move-target-' + fromTeam + '-' + playerKey;
  const sel = document.getElementById(selId);
  if(!sel) return;
  const toTeam = sel.value;
  if(toTeam === fromTeam) { alert('目标球队与当前球队相同'); return; }

  // Find player data
  let playerData = null;
  let playerIndex = -1;
  TEAMS_DATA.forEach(t => {
    if(t.team === fullTeamName(fromTeam)) {
      t.players.forEach((p, i) => {
        if((p.name_en || p.name) === playerKey) {
          playerData = { ...p };
          playerIndex = i;
        }
      });
    }
  });

  if(!playerData) { alert('未找到球员数据'); return; }

  // Remove from source team
  TEAMS_DATA.forEach(t => {
    if(t.team === fullTeamName(fromTeam)) {
      t.players.splice(playerIndex, 1);
    }
  });

  // Add to target team
  TEAMS_DATA.forEach(t => {
    if(t.team === fullTeamName(toTeam)) {
      t.players.push(playerData);
    }
  });

  adminAddLog(`🔄 移动球员：${playerData.name} ${fromTeam} → ${toTeam}`);
  alert(`✅ ${playerData.name} 已从 ${fromTeam} 移动到 ${toTeam}`);
  adminSearchMovePlayer(); // refresh
}
